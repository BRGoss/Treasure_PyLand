def accept_thirty(sentence):
	words = sentence.split(' ')
	theString = ''
	count = 0
	i = 0
	while (count < 30):
		count = count + len(words[i]) + 1
		if (count <= 30):
			theString = theString + words[i]
			if((count + len(words[i + 1]) + 1) <= 30):
				theString = theString + ' '
		i = i + 1
	return theString
